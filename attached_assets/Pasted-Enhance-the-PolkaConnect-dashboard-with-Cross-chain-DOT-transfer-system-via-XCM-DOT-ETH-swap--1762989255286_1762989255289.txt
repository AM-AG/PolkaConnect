"use client";
import { useState } from "react";
import { Card, CardHeader, CardTitle, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Select, SelectTrigger, SelectContent, SelectItem } from "@/components/ui/select";
import { Input } from "@/components/ui/input";
import { toast } from "sonner";
import { sendCrossChainTransfer } from "@/app/api/xcm";

export default function TransferPage() {
  const [fromChain, setFromChain] = useState("polkadot");
  const [toChain, setToChain] = useState("moonbeam");
  const [amount, setAmount] = useState("");
  const [destination, setDestination] = useState("");

  const handleTransfer = async () => {
    try {
      await sendCrossChainTransfer({ fromChain, toChain, amount, destination });
      toast.success(`Transfer of ${amount} DOT to ${toChain} initiated.`);
    } catch (e) {
      toast.error("Transfer failed: " + e.message);
    }
  };

  return (
    <Card>
      <CardHeader><CardTitle>Cross-Chain Transfer</CardTitle></CardHeader>
      <CardContent className="space-y-3">
        <Select onValueChange={setFromChain}>
          <SelectTrigger>{fromChain}</SelectTrigger>
          <SelectContent>
            <SelectItem value="polkadot">Polkadot</SelectItem>
          </SelectContent>
        </Select>

        <Select onValueChange={setToChain}>
          <SelectTrigger>{toChain}</SelectTrigger>
          <SelectContent>
            <SelectItem value="moonbeam">Moonbeam</SelectItem>
            <SelectItem value="astar">Astar</SelectItem>
            <SelectItem value="acala">Acala</SelectItem>
          </SelectContent>
        </Select>

        <Input placeholder="Amount (DOT)" value={amount} onChange={(e) => setAmount(e.target.value)} />
        <Input placeholder="Destination Address" value={destination} onChange={(e) => setDestination(e.target.value)} />

        <Button className="w-full bg-pink-600" onClick={handleTransfer}>Send Transfer</Button>
      </CardContent>
    </Card>
  );
}

File: /app/api/xcm.ts

Implement XCM transfer logic using @polkadot/api.

import { ApiPromise, WsProvider } from "@polkadot/api";
import { web3FromAddress } from "@polkadot/extension-dapp";

const CHAINS = {
  moonbeam: { paraId: 2004 },
  astar: { paraId: 2006 },
  acala: { paraId: 2000 },
};

export async function sendCrossChainTransfer({ fromChain, toChain, amount, destination }) {
  const provider = new WsProvider("wss://rpc.polkadot.io");
  const api = await ApiPromise.create({ provider });

  const injector = await web3FromAddress(destination);
  const destParaId = CHAINS[toChain]?.paraId || 2004;

  const dest = { V1: { parents: 0, interior: { X1: { Parachain: destParaId } } } };
  const beneficiary = { V1: { parents: 0, interior: { X1: { AccountId32: { network: 'Any', id: destination } } } } };
  const assets = [{ id: { Concrete: { parents: 0, interior: { Here: null } } }, fun: { Fungible: BigInt(amount * 10 ** 10) } }];

  const tx = api.tx.polkadotXcm.send(dest, beneficiary, assets, 0);
  const sender = (await web3FromAddress(destination)).signer;

  await tx.signAndSend(destination, { signer: sender }, ({ status }) => {
    console.log("Transaction status:", status.type);
  });
}

ğŸ” 3ï¸âƒ£ Live Transactions Feed
File: /app/transactions/page.tsx

Subscribe to Polkadot block headers and show latest transactions.

import { ApiPromise, WsProvider } from "@polkadot/api";
import { useEffect, useState } from "react";

export default function TransactionsPage() {
  const [txs, setTxs] = useState([]);

  useEffect(() => {
    async function subscribe() {
      const provider = new WsProvider("wss://rpc.polkadot.io");
      const api = await ApiPromise.create({ provider });

      await api.rpc.chain.subscribeNewHeads(async (header) => {
        const block = await api.rpc.chain.getBlock(header.hash);
        const extrinsics = block.block.extrinsics.map((ext) => ext.method.section + "." + ext.method.method);
        setTxs((prev) => [...extrinsics.slice(0, 5), ...prev].slice(0, 50));
      });
    }
    subscribe();
  }, []);

  return (
    <div className="p-4">
      <h2 className="text-xl font-bold mb-2">Live Polkadot Transactions</h2>
      <ul className="font-mono text-sm space-y-1">
        {txs.map((tx, i) => <li key={i}>{tx}</li>)}
      </ul>
    </div>
  );
}

ğŸ’³ /sdk/js/wallet.ts
import { web3Enable, web3Accounts } from '@polkadot/extension-dapp';
import Web3 from 'web3';

export async function connectWallet(chain: 'polkadot' | 'ethereum') {
  if (chain === 'polkadot') {
    const extensions = await web3Enable('PolkaConnect');
    if (extensions.length === 0) throw new Error('No PolkadotJS extension found');
    const accounts = await web3Accounts();
    return { chain, accounts };
  }

  if (chain === 'ethereum') {
    if (!window.ethereum) throw new Error('MetaMask not found');
    const provider = new Web3(window.ethereum);
    await window.ethereum.request({ method: 'eth_requestAccounts' });
    const accounts = await provider.eth.getAccounts();
    return { chain, accounts };
  }

  throw new Error('Unsupported chain');
}

export async function getWalletBalance(chain: string, address: string) {
  const endpoint = `/api/balance?chain=${chain}&address=${address}`;
  const response = await fetch(endpoint);
  return response.json();
}

export async function swapTokens(from: string, to: string, amount: number) {
  const res = await fetch('/api/swap', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ from, to, amount }),
  });
  return res.json();
}
